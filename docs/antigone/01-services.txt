# -*- mode: sh; -*-

# Services
# ========

zfs create          \
    system/services

mkdir -p /var/services

# Podman containers
# ~~~~~~~~~~~~~~~~~

zfs create system/podman

# Quassel
# ~~~~~~~

zfs create                  \
    -o acltype=posixacl     \
    -o mountpoint=legacy    \
    system/services/quassel

mkdir -p /var/services/quassel
mount -t zfs system/services/quassel /var/services/quassel
