# -*- mode: sh; -*-

# Services
# ========

zfs create          \
    system/services

mkdir -p /var/services

# nginx
# ~~~~~~~

zfs create                  \
    -o mountpoint=legacy    \
    system/services/nginx

mkdir -p /var/services/nginx
mount -t zfs system/services/nginx /var/services/nginx

mkdir -p /var/services/nginx/acme
chown nobody:60 /var/services/nginx/acme # 60 is nginx gid

# actual-budget
# ~~~~~

zfs create                        \
    system/services/actual-budget

zfs create                                 \
    -o mountpoint=legacy                   \
    system/services/actual-budget/tancredi

mkdir -p /var/services/actual-budget/tancredi
mount -t zfs                                 \
      system/services/actual-budget/tancredi \
      /var/services/actual-budget/tancredi
